Card

Instructions for using the Card component.

Description

Used to create a card view with image, title, description with additional labels and a button

Usage


import React, { Component } from 'react';
import {
  View,
} from 'react-native';

import Card from './src/lib/card/card';

const image1 = require('./src/assets/product.png');
const image2 = require('./src/assets/product2.png');

export default class App extends Component<{}> {
  render() {
    return (
      <View style={{ backgroundColor: 'white', flex: 1, flexDirection: 'row' }}>
        <Card
          buttonLabel="99%"
          description="It helps to outline the It helps to outline the"
          highlightedLabel="500.00 $"
          imageSource={image2}
          strikedLabel="799.00 $"
          title="Product #2"
        />
        <Card
          buttonLabel="99%"
          description="It helps to outline the It helps to outline the"
          highlightedLabel="500.00 $"
          imageSource={image1}
          strikedLabel="799.00 $"
          title="Product #1"
        />
      </View>
    );
  }
}
