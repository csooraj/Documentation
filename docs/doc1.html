Thumbnail

Instructions for using the thumbnail component.

Description

Used to create a thumbnail image either with text or buttons or whatever child component that you want to display.

Usage

import React, { Component } from 'react';
import {
  Platform,
  StyleSheet,
  WebView,
  Text,
  View
} from 'react-native';

import Styles from './styles';
import Button from '../button/button';
import Thumbnail from '../thumbnail/thumbnail';

export default class App extends Component<Props> {

  render() {
    return (
      <View>

       // Large Thumbnail with remote image and text align on left end

        <Thumbnail
          thumbLabel="CATEGORY 1"
          textOnly={true}
          thumbNailStyle={Styles.category1}
          thumbImageStyle={Styles.category1Image}
          thumbLabelStyle={Styles.category1Label}
          source={{uri: 'https://cdn.pixabay.com/photo/2017/09/07/17/27/background-2726039_960_720.png'}}
        />


        // Small Thumbnail with local image and text align on center

        <Thumbnail
          thumbLabel="CATEGORY 2"
          textOnly={true}
          thumbNailStyle={Styles.category2}
          thumbImageStyle={Styles.category2Image}
          thumbLabelStyle={Styles.category2Label}
          source={require('../../assets/test.png')}
        />

        // Large Thumbnail with other components inside

        <Thumbnail
          textOnly={false}
          thumbNailStyle={Styles.category3}
          thumbImageStyle={Styles.category3Image}
          thumbLabelStyle={Styles.category3Label}
          source={require('../../assets/newyear.png')}
        >
         <View style={Styles.containerView}>
           <View style={Styles.childContainer}>
             <Text
               adjustsFontSizeToFit
               style={Styles.category3Label}
             >
              Happy New Year!
             </Text>
             <Button
               style={Styles.visitButton}
               label="Visit"
             />
           </View>
         </View>
        </Thumbnail>
      </View>
    );
  }
}

Reference

source

source of the image, provide either path of local image or url of the image (required)

TypeDefault
objectnone

textOnly

boolean indicating the type of thumbnail. For mixed components rendering set this false and set true for rendering only text (required)

TypeDefault
booleantrue

label

label to be displayed on the thumbnail (optional)

TypeDefault
stringnone

thumbNailStyle

Outer container style of the thumbnail (optional).You can use all the styles supported by the touchable opacity component in react native.

TypeDefault
object (style){ height: 100, width: 250, margin: 10, alignSelf: 'center'}

thumbImageStyle

Style of the image background (optional). You can use all the styles supported by image background component in react native.

TypeDefault
object (style){ width: '100%',height: '100%',borderColor: '#000', borderWidth: 1, alignItems: 'flex-start',justifyContent: 'flex-end', borderRadius: 5}

thumbLabelStyle

Style of the title rendered over the image (optional). You can use all the styles supported by the text component in react native.

TypeDefault
object (style){ color: 'white', margin: '3%', fontSize: 18 }
